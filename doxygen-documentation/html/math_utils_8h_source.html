54inline V3D rotateVec(const V3D &v, double alpha, const V3D &axis) {
55 assert(IS_EQUAL(axis.squaredNorm(), 1));
56
57 double cosa = cos(alpha);
58 double sina = sin(alpha);

60 return v * cosa + axis.cross(v) * sina + axis * (axis.dot(v)) * (1 - cosa);

66inline double angleBetween(const V3D &u, const V3D &v) {
67 // U.V = |U|*|V|*cos(angle)

96inline double getRandomGaussian() {
97 double x1, x2, rquad;
98 do {
99 x1 = 2.0 * getRandomNumberInRange(0, 1) - 1.0;
100 x2 = 2.0 * getRandomNumberInRange(0, 1) - 1.0;
101 rquad = x1 * x1 + x2 * x2;
102 } while (rquad >= 1 || rquad <= 0);
103
104 double fac = sqrt(-2 * log(rquad) / rquad);
105
106 return fac * x2;
107}

140inline double getRotationAngle(const Quaternion &q, const V3D &v) {
141 int sinSign = SGN(q.vec().dot(v));
142 double result = 2 * safeACOS(q.w());
143 if (sinSign < 0)
144 result = -result;
145 if (result > PI)
146 result -= 2 * PI;
147 if (result < -PI)
148 result += 2 * PI;
149 return result;
150}

195inline void getVectorsOrthogonalTo(const V3D &v, V3D &a, V3D &b) {
196 // try to choose a vector in the y-z plane, if the z-coordinate is
197 // significant enough
198 if (v[0] * v[0] / (v.squaredNorm()) < 0.5) // if the x component of the current vector is not too large,
199 // relatively speaking, then choose x as the main component for a
200 // (they won't be aligned perfectly)
201 a = V3D(1, 0, 0);
202 else // if the x component of the current vector is large, then the y (or
203 // z) components cannot be the only non-zero values, so a is not
204 // conlinear with the current vector
205 a = V3D(0, 1, 0);
206 b = v.cross(a); // b is orthogonal to both the current vector and a
207 a = b.cross(v); // and a is orthogonal to both a and this...
208 a.normalize();
209 b.normalize();
210}

236inline void computeEulerAxesFromQuaternion(const Quaternion &q, V3D &a, V3D &b, V3D &c) {
237 // assume q is a rotation transforming vectors from coordinate frame A to B
238 // (i.e. B_q_A). We want a sequence of Euler rotations that take you from B
239 // to A (i.e. B_q_A = rot(c) * rot(b) * rot(a). There are many options, so
240 // just choose one of them...
241 a = q.vec().normalized();
242 // then b and c should be orthogonal to a - find a rotation between a and
243 // the x-axis, and then apply the same rotation to the y and z axes to get b
244 // and c...
245 V3D rotAxis = a.cross(V3D(1, 0, 0)).normalized();
246 double rotAngle = angleBetween(a, V3D(1, 0, 0));
247 if (rotAxis.norm() < 0.000001) {
248 rotAxis = a = V3D(1, 0, 0);
249 rotAngle = 0;
250 }
251
252 // V3D testV = V3D(1,0,0).rotate(-rotAngle, rotAxis);
253 assert((rotateVec(V3D(1, 0, 0), -rotAngle, rotAxis) - a).norm() < 0.000001);
254
255 a = rotateVec(V3D(1, 0, 0), -rotAngle, rotAxis);
256 b = rotateVec(V3D(0, 1, 0), -rotAngle, rotAxis);
257 c = rotateVec(V3D(0, 0, 1), -rotAngle, rotAxis);
258}

264inline void computeEulerAnglesFromQuaternion(const Quaternion &q, const V3D &a, const V3D &b, const V3D &c, double &alpha, double &beta, double &gamma) {
265 // the idea here is that the a axis only gets rotated about b and c, which
266 // are assumed to be orthogonal to each other. Based on this info, we can
267 // first compute the angles beta and gamma
268 assert(IS_ZERO(a.dot(b)) && IS_ZERO(b.dot(c)));
269 assert(IS_ZERO(a.norm() - 1) && IS_ZERO(b.norm() - 1) && IS_ZERO(c.norm() - 1));
270
271 V3D aRot = q * a;
272
273 if (IS_ZERO(a.dot(c))) {
274 bool circular = a.cross(b).dot(c) > 0;
275 // the three axes form an orthonormal basis (i.e. Tait-Bryan)...
276 // singularity around beta = -PI/2 or PI/2
277 if (circular) {
278 beta = -safeASIN(aRot.dot(c));
279 gamma = atan2(aRot.dot(b), aRot.dot(a));
280 } else {
281 beta = safeASIN(aRot.dot(c));
282 gamma = atan2(-aRot.dot(b), aRot.dot(a));
283 }
284 } else if (IS_ZERO(a.dot(c) - 1)) {
285 // these are "proper" euler axes, where the first and the last one are
286 // the same... singularity around beta = 0 or PI
287 V3D lastAxis = a.cross(b);
288 beta = safeACOS(aRot.dot(a));
289 gamma = atan2(aRot.dot(b), -aRot.dot(lastAxis));
290 } else {
291 // dunno what this is.... freak out...
292 alpha = beta = gamma = 0;
293 assert(false);
294 return;
295 }
296
297 Quaternion qLeft = getRotationQuaternion(-beta, b) * getRotationQuaternion(-gamma, c) * q;
298 alpha = getRotationAngle(qLeft, a);
299
300 // Quaternion residual = (getRotationQuaternion(gamma, c) *
301 // getRotationQuaternion(beta, b) * getRotationQuaternion(alpha, a) *
302 // q.getComplexConjugate()); printf("residual: %lf (%lf %lf %lf %lf)
303 // --> %lf %lf %lf\n", residual.v.length(), residual.s, residual.v[0],
304 // residual.v[1], residual.v[2], alpha, beta, gamma);
305 assert(IS_ZERO((getRotationQuaternion(gamma, c) * getRotationQuaternion(beta, b) * getRotationQuaternion(alpha, a) * q.inverse()).vec().norm() / 10e5));
306}

315inline bool sameRotation(const Quaternion &q1, const Quaternion &q2) {
316 // we know that q and -q represent the same rotation, so take that into
317 // account...
318 if (q1.isApprox(q2, 1e-7))
319 return true;
320 Quaternion qTmp = q2;
321 qTmp.w() *= -1;
322 qTmp.vec() *= -1;
323 return q1.isApprox(qTmp, 1e-7);
324}
