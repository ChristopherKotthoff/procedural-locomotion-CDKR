12class SwingPhaseContainer {
13public:
14 // this is the limb the swing phase is applied to...
15 std::shared_ptr<RobotLimb> limb;
16
17 // keep track of when a contact should be broken (e.g. a foot lifts), and
18 // when it should be established again - the swing phase is defined to take
19 // place between these two discrete events...
20 DynamicArray<pair<double, double>> swingPhases;
21
22public:
23 SwingPhaseContainer(const std::shared_ptr<RobotLimb> &l) {
24 limb = l;
25 }
26
27 void addSwingPhase(double start, double end) {
28 // the swing phase must end after it begins!
29 if (start >= end) {
30 assert(false);
31 return;
32 }
33 swingPhases.push_back(pair<double, double>(start, end));
34 }
35

37 while (swingPhases.size() > 0 && swingPhases.front().second < t)

42class PeriodicGait {
43public:
44 // this is the duration of the stride, in seconds...
45 double strideDuration = 1.0;
46
47 // trajectories that control variations in body motion trajectories, encoded
48 // as a function of the stride phase
49 Trajectory1D bodyHeightOffset, bodyForwardLeanOffset, bodySidewaysLeanOffset;
50 Trajectory1D bodyPitchOffset, bodyRollOffset, bodyYawOffset;
51
52 // and the swing period is represented here relative to the overall phase of
53 // the gait cycle... e.g. starts at 0.25 and ends at 0.4, where the gait
54 // cycle is normalized to go from 0 to 1.
55 DynamicArray<SwingPhaseContainer> swingPhases;
56
57public:
58 PeriodicGait() {
59 clearAllBodyOffsetTrajectories();
60 }
61
62 void addSwingPhaseForLimb(const std::shared_ptr<RobotLimb> &l, double start, double end) {
63 // the following must hold for a swing phase
64 // 1) it must start before it ends
65 // 2) the interval (start, end) must have a non-zero intersection
66 // with (0, 1), so either end is greater than 0 (and start can be either
67 // positive or negative), or start is less than 1 (and end can be less
68 // than or greater than 1) 3) end-start must be less than or equal
69 // to 1 ( if equal to 1 this means the limb is always in swing)
70 if (!(start < end)) {
71 assert(false);
72 return;
73 }
74 if (!(end > 0 || start < 1)) {
75 assert(false);
76 return;
77 }
78 if (!(end - start)) {
79 assert(false);
80 return;
81 }
82
83 // periodic gaits will admit only one swing phase per gait cycle...
84 for (auto ls : swingPhases)
85 if (ls.limb == l) {
86 assert(false);
87 return;
88 }
89
90 swingPhases.push_back(SwingPhaseContainer(l));
91 swingPhases.back().addSwingPhase(start, end);
92 }
93
94 void clearAllBodyOffsetTrajectories() {
95 bodyHeightOffset.clear();
96 bodyHeightOffset.addKnot(0, 0);
97 bodyHeightOffset.addKnot(1, 0);
98
99 bodyForwardLeanOffset.clear();
100 bodyForwardLeanOffset.addKnot(0, 0);
101 bodyForwardLeanOffset.addKnot(1, 0);
102
103 bodySidewaysLeanOffset.clear();
104 bodySidewaysLeanOffset.addKnot(0, 0);
105 bodySidewaysLeanOffset.addKnot(1, 0);
106
107 bodyPitchOffset.clear();
108 bodyPitchOffset.addKnot(0, 0);
109 bodyPitchOffset.addKnot(1, 0);
110
111 bodyRollOffset.clear();
112 bodyRollOffset.addKnot(0, 0);
113 bodyRollOffset.addKnot(1, 0);
114
115 bodyYawOffset.clear();
116 bodyYawOffset.addKnot(0, 0);
117 bodyYawOffset.addKnot(1, 0);
118 }

179class ContactSchedule {
180public:
181 // swing phases stored in absolute time...
182 DynamicArray<SwingPhaseContainer> cs;
183
184public:
185 void addSwingPhaseForLimb(const std::shared_ptr<RobotLimb> &l, double start, double end) {
186 if (!(start < end)) {
187 assert(false);
188 return;
189 }
190 if (!(end > 0 || start < 1)) {
191 assert(false);
192 return;
193 }
194 if (!(end - start)) {
195 assert(false);
196 return;
197 }
198
199 int lIndex = 0;
200
201 // periodic gaits will admit only one swing phase per gait cycle...
202 for (auto ls : cs) {
203 if (ls.limb == l)
204 break;
205 lIndex++;
206 }
207
208 if (lIndex == (int)cs.size())
209 cs.push_back(SwingPhaseContainer(l));
210
211 // now, make sure that this new swing phase starts after the previous
212 // one has ended...
213 if (cs[lIndex].swingPhases.size() > 0) {
214 // if the new swing phase starts right when the previous one ends,
215 // merge them...
216 if (fabs(cs[lIndex].swingPhases.back().second - start) < 0.0001) {
217 cs[lIndex].swingPhases.back().second = end;
218 return;
219 }
220 if (cs[lIndex].swingPhases.back().second > start) {
221 assert(false);
222 return;
223 }
224 }
225
226 cs[lIndex].addSwingPhase(start, end);
227 }
228
229 void addPeriodicGaitToContactSequence(const PeriodicGait &pg, double startTime) {
230 for (auto ls : pg.swingPhases)
231 addSwingPhaseForLimb(ls.limb, startTime + ls.swingPhases.front().first * pg.strideDuration,
232 startTime + ls.swingPhases.front().second * pg.strideDuration);
233 }
234
235 SwingPhaseContainer *getSwingPhaseContainerForLimb(const std::shared_ptr<RobotLimb> &l) {
236 for (uint i = 0; i < cs.size(); i++)
237 if (cs[i].limb == l)
238 return &cs[i];
239 return nullptr;
240 }
241
247 ContactPhaseInfo getContactPhaseInformation(const std::shared_ptr<RobotLimb> &l, double t) {
248 SwingPhaseContainer *spc = nullptr;
249 for (uint i = 0; i < cs.size(); i++)
250 if (cs[i].limb == l) {
251 spc = &cs[i];
252 break;
253 }
254
255 // if we have no information for this limb, we assume no one gave it any
256 // swing phases, so it must be allways in stance then...
257 if (spc == nullptr)
258 return ContactPhaseInfo(false, 100, 50);
259
260 // if this limb has been registered but it has no swing phases after t,
261 // then it should be assumed it will remain in stance for the forseable
262 // future...
263 if (spc->swingPhases.size() == 0)
264 return ContactPhaseInfo(false, 100, 50);
265
266 // now, t might be before any of the swing phases are starting...
267 if (spc->swingPhases.front().first > t)
268 return ContactPhaseInfo(false, spc->swingPhases[0].first - t, spc->swingPhases[0].first - t);
269
270 // or it could be after the last swing phase ends...
271 if (spc->swingPhases.back().second < t)
272 return ContactPhaseInfo(false, 100, 50);
273
274 // if none of the above are true, t must fall between swing phases, or
275 // within a swing phase...
276 for (int i = 0; i < (int)spc->swingPhases.size(); i++) {
277 // see if t falls within a swing phase
278 if (t >= spc->swingPhases[i].first && t <= spc->swingPhases[i].second)
279 return ContactPhaseInfo(true, spc->swingPhases[i].second - spc->swingPhases[i].first, spc->swingPhases[i].second - t);
280 // since swing phases are sorted, if t is smaller than the next
281 // swing phase, it means it falls within a contact region... note:
282 // we know that t does not start after the last swing phase, so the
283 // "i+1" here should be safe...
284 if (t <= spc->swingPhases[i + 1].first)
285 return ContactPhaseInfo(false, spc->swingPhases[i + 1].first - spc->swingPhases[i].second, spc->swingPhases[i + 1].first - t);
286 }
287
288 // all right, we should never get here...
289 assert(false);
290 // but the show must go on...
291 return ContactPhaseInfo(false, 100, 50);
292 }
