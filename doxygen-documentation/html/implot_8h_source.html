341// Range defined by a min/max value.
342struct ImPlotRange {
343 double Min, Max;
344 ImPlotRange() { Min = 0; Max = 0; }
345 ImPlotRange(double _min, double _max) { Min = _min; Max = _max; }
346 bool Contains(double value) const { return value >= Min && value <= Max; }
347 double Size() const { return Max - Min; }
348 double Clamp(double value) const { return (value < Min) ? Min : (value > Max) ? Max : value; }

351// Combination of two range limits for X and Y axes. Also an AABB defined by Min()/Max().
352struct ImPlotRect {
353 ImPlotRange X, Y;
354 ImPlotRect() { }
355 ImPlotRect(double x_min, double x_max, double y_min, double y_max) { X.Min = x_min; X.Max = x_max; Y.Min = y_min; Y.Max = y_max; }
356 bool Contains(const ImPlotPoint& p) const { return Contains(p.x, p.y); }
357 bool Contains(double x, double y) const { return X.Contains(x) && Y.Contains(y); }
358 ImPlotPoint Size() const { return ImPlotPoint(X.Size(), Y.Size()); }
359 ImPlotPoint Clamp(const ImPlotPoint& p) { return Clamp(p.x, p.y); }
360 ImPlotPoint Clamp(double x, double y) { return ImPlotPoint(X.Clamp(x),Y.Clamp(y)); }
361 ImPlotPoint Min() const { return ImPlotPoint(X.Min, Y.Min); }
362 ImPlotPoint Max() const { return ImPlotPoint(X.Max, Y.Max); }
363};

496// ImPlot::PlotLine(...);

498// EndPlot();
499// }
500// }
501// EndSubplots();
502// }

505//
506// [0] | [1] | [2]

511//
512// - #title_id must be unique to the current ImGui ID scope. If you need to avoid ID
513// collisions or don't want to display a title in the plot, use double hashes
514// (e.g. "MySubplot##HiddenIdText" or "##NoTitle").
515// - #rows and #cols must be greater than 0.
516// - #size is the size of the entire grid of subplots, not the individual plots
517// - #row_ratios and #col_ratios must have AT LEAST #rows and #cols elements,
518// respectively. These are the sizes of the rows and columns expressed in ratios.
519// If the user adjusts the dimensions, the arrays are updated with new ratios.
520//
521// Important notes regarding BeginPlot from inside of BeginSubplots:
522//
523// - The #title_id parameter of _BeginPlot_ (see above) does NOT have to be
524// unique when called inside of a subplot context. Subplot IDs are hashed
525// for your convenience so you don't have call PushID or generate unique title
526// strings. Simply pass an empty string to BeginPlot unless you want to title
527// each subplot.
528// - The #size parameter of _BeginPlot_ (see above) is ignored when inside of a
529// subplot context. The actual size of the subplot will be based on the
530// #size value you pass to _BeginSubplots_ and #row/#col_ratios if provided.
