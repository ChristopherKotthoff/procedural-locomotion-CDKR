163 // vec3 -> quat -> trackbalTransforms -> quat -> vec3
165 bool getTransforms(quat& q, const char* label, vec3& dir, float size) {
166 const float len = length(dir);
167 q = angleAxis(acosf(dir.x/len), normalize(vec3(0.0f, -dir.z, dir.y)));
168
169 bool ret = drawFunc(label, size);
170 if (ret) dir = (q * vec3(1.0f, 0.0f, 0.0f)) * len ; //return vector with original lenght
171
172 return ret;
173 }
174 // Vec4 (xyz axis, w angle) -> quat -> trackbalTransforms -> quat -> vec4
176 bool getTransforms(quat& q, const char* label, vec4& axis_angle, float size) {
177 q = angleAxis(axis_angle.w,vec3(axis_angle)); //g.ConvertFromAxisAngle();
178
179 bool ret = drawFunc(label, size);
180 if (ret) axis_angle = vec4(vec3(axis(q)),angle(q));
181
182 return ret;
183 }
