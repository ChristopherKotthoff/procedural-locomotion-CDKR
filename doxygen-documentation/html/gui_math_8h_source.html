36inline glm::mat4 getGLMTransform(const V3D &scale, const Quaternion &orientation, const P3D &position) {
37 // scale, then rotate, then translate...
38 glm::mat4 transform = glm::mat4(1.0);
39 AngleAxisd rot(orientation);
40 transform = glm::translate(transform, toGLM(position));
41 transform = transform * glm::rotate(glm::mat4(1.0), (float)(rot.angle()), toGLM(rot.axis()));
42 transform = transform * glm::scale(glm::mat4(1.0), toGLM(scale));
43 return transform;
44}
