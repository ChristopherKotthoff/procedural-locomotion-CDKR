17inline std::vector<std::string> split(const std::string &s, const char delim) {
18 std::vector<std::string> v;
19 auto p = std::begin(s);
20 for (auto q = std::find(p, std::end(s), delim); q != std::end(s); q = std::find(++p, std::end(s), delim)) {
21 v.emplace_back(p, q);
22 p = q;
23 }
24 if (p != std::end(s))
25 v.emplace_back(p, std::end(s));
26 return v;
27}
