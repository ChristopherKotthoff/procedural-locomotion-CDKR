51public:
52 explicit RealTimeLinePlot2D(const std::string &title, const std::string &xlabel, const std::string &ylabel, int maxSize = 1000, int plotHeight = 300)
53 : title(title), xlabel(xlabel), ylabel(ylabel), maxSize(maxSize), plotHeight(plotHeight) {
54 if (maxSize > 2000) // for saving memory...
55 maxSize = 2000;

79 if (data.size() < maxSize) {

82 data[offset] = {x, y};
83 offset = (offset + 1) % maxSize;

91 void draw() {
92 ImGui::Checkbox(std::string("Always Fit X Axes##" + title).c_str(), &fitAxisX);
93 ImGui::SameLine();
94 ImGui::Checkbox(std::string("Always Fit Y Axes##" + title).c_str(), &fitAxisY);
95 // use slider for fit axis x instead of ImPlot API.
96 if (fitAxisY) {
97 ImPlot::SetNextAxisToFit(ImAxis_Y1);
98 }
99 if (fitAxisX) {
100 ImGui::SliderFloat(std::string("X Axis Scale##" + title).c_str(), &scaleAxisX, 0.1, 1);
101 ImPlot::SetNextAxisLimits(ImAxis_X1, getXBegin() * scaleAxisX + getXEnd() * (1.f - scaleAxisX), getXEnd(), ImGuiCond_Always);
102 }
103
104 // draw plot
105 if (ImPlot::BeginPlot(title.c_str(), ImVec2(-1, plotHeight))) {
106 ImPlot::SetupAxis(ImAxis_X1, xlabel.c_str());
107 ImPlot::SetupAxis(ImAxis_Y1, ylabel.c_str());
108 for (DRAWINDEX_ = 0; DRAWINDEX_ < lineSpecs.size(); DRAWINDEX_++) {
109 ImPlot::SetNextMarkerStyle(lineSpecs[DRAWINDEX_].marker);
110 ImPlot::PlotLineG(
111 lineSpecs[DRAWINDEX_].label.c_str(),
112 [](void *data, int idx) {
113 auto *my_data = (RealTimeLinePlot2D<T> *)data;
114 auto &datapoint = my_data->getData(idx);
115 ImPlotPoint p;
116 p.x = (float)datapoint.first;
117 p.y = my_data->lineSpecs[my_data->DRAWINDEX_].getter(datapoint.second);
118 return p;
119 },
120 this, getSize());
121 }
122 ImPlot::EndPlot();
123 }
124
125 ImGui::Separator();
126 }
127
128private:
129 int getSize() const {
130 if (data.size() < maxSize)
131 return data.size();
132 return maxSize;
133 }
134
135 int getOffset() const {
136 return offset;
137 }
138
139 const std::pair<float, T> &getData(int idx) {
140 if (data.size() < maxSize)
141 return data[idx];
142 return data[(offset + idx) % maxSize];
143 }
144
145 float getXBegin() const {
146 if (data.size() == 0)
147 return -1;
148 if (data.size() < maxSize)
149 return data[0].first;
150 return data[offset].first;
151 }
152
153 float getXEnd() const {
154 if (data.size() == 0)
155 return 1;
156 if (data.size() < maxSize)
157 return data.back().first;
158 return data[offset - 1].first;
159 }
160};
