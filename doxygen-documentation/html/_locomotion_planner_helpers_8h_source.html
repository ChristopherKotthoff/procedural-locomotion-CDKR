113 //given world coordinates for the step locations, generate continuous trajectories for each of a robot's feet
114 Trajectory3D generateLimbTrajectory(const std::shared_ptr<LeggedRobot> robot, int limbIndex, const LimbMotionProperties& lmp, double tStart, double tEnd, double dt) {
115 const std::shared_ptr<RobotLimb>& limb = robot->getLimb(limbIndex);
116 double t = tStart;
117 Trajectory3D traj;
118
119 //always start the trajectory from the current location of the robot
120 V3D startingEEPos = V3D(limb->getEEWorldPos());
121 traj.addKnot(t, startingEEPos);
122
123 t += dt;
124
125 while (t < tEnd) {
126 ContactPhaseInfo cpi = cpm->getCPInformationFor(limb, t);
127
128 if (cpi.isStance()) {
129 double tEndOfStance = t + cpi.getTimeLeft();
130 // in stance, we want the foot to not slip, while keeping to
131 // the ground...
132 V3D eePos = traj.getKnotValue(traj.getKnotCount() - 1);
133 double groundHeight = ground.get_height(eePos[0], eePos[2]);
134 eePos.y() = groundHeight + limb->ee->radius * lmp.contactSafetyFactor; // account for the size of the ee
135 while (t <= tEndOfStance && t < tEnd) {
136 traj.addKnot(t, eePos);
137 t += dt;
138 }
139 } else {
140 double tEndOfSwing = t + cpi.getTimeLeft();
141
142 V3D finalEEPos = V3D(getCurrentOrUpcomingPlannedContactLocation(limb, tEndOfSwing));
143
144 // when we first transition to a swing phase (at some time
145 // sample t_i), this could be at the very beginning of it,
146 // or it could have been a "little while ago" and so the
147 // position of the swing foot at this moment in time needs
148 // to account for this sampling-induced variation...
149 bool firstTimeStepInSwingPhase = (cpi.getDuration() - cpi.getTimeLeft()) < dt;
150
151 // plan motion for the entire swing phase here to avoid
152 // computing redundant information...
153 while (t <= tEndOfSwing) {
154 ContactPhaseInfo cpiSwing = cpm->getCPInformationFor(limb, t);
155
156 double factor = 1.0;
157 if (firstTimeStepInSwingPhase)
158 factor = (cpi.getDuration() - cpi.getTimeLeft()) / dt;
159 firstTimeStepInSwingPhase = false;
160
161 V3D oldEEPos = traj.getKnotValue(traj.getKnotCount() - 1);
162 // now, we have the remainder of the swing phase to go
163 // from the old step position to the final stepping
164 // location. Based on this we know how much we should be
165 // travelling over a time window dt...
166 double dTimeStep = 1.0;
167 // the -0.05 thing here means we want the swing foot to
168 // reach its target 0.05s before the swing phase ends
169 if (cpiSwing.getTimeLeft() - 0.05 > dt)
170 dTimeStep = dt / (cpiSwing.getTimeLeft() - 0.05) * factor;
171
172 V3D deltaStep = dTimeStep * (finalEEPos - oldEEPos);
173 V3D eePos = oldEEPos + deltaStep;
174 double groundHeight = ground.get_height(eePos[0], eePos[2]);
175 // add ground height + ee size as offset...
176 eePos.y() = groundHeight + lmp.swingFootHeightTraj.evaluate_linear(cpiSwing.getPercentageOfTimeElapsed()) * lmp.swingFootHeight +
177 lmp.swingHeightOffsetTrajDueToFootSize.evaluate_linear(cpiSwing.getPercentageOfTimeElapsed()) * limb->ee->radius;
178
179 traj.addKnot(t, eePos);
180 t += dt;
181 }
182 }
183 }
184
185 return traj;
186 }

190 The body frame motion represents the *average* linear trajectory and heading for a robot's trunk.
191 Roll, pitch and parasitic components of motion (e.g. transient fluctuations in yaw / translation
192 of COM arising from reactions to small perturbations or planned body sway, e.g. side to side motion during a
193 stride) are not captured by the motion of the body frame.
194 */

201 void generate(const bFrameState& startingbFrameState) {
202 double headingAngle = startingbFrameState[3];
203 P3D pos(startingbFrameState[0], startingbFrameState[1], startingbFrameState[2]);
204 double vForward = targetForwardSpeed;
205 double vSideways = targetSidewaysSpeed;
206 double turningSpeed = targetTurngingSpeed;
207
208 bFramePosTrajectory.clear();
209 bFrameHeadingTrajectory.clear();
210 bFrameVels.clear();
211
212 double t = tStart;
213
214 // generate trajectories that capture the motion of the robot's body frame...
215 while (t < tEnd) {
216 Quaternion heading = getRotationQuaternion(headingAngle, V3D(0, 1, 0));
217
218 bFramePosTrajectory.addKnot(t, V3D(pos));
219 bFrameHeadingTrajectory.addKnot(t, headingAngle);
220
221 bFrameVels.addKnot(t, V3D(vForward, vSideways, turningSpeed));
222
223 vForward = targetForwardSpeed;
224 vSideways = targetSidewaysSpeed;
225 turningSpeed = targetTurngingSpeed;
226
227 // TODO: Trajectory Planning
228 //
229 // update "pos" and "headingAngle".
230 // compute nextstep's "pos" and "headingAngle" by integrating "vForward", "vSideways" and "turningSpeed".
231 //
232 // Hints:
233 // - quaternion heading = Roty(headingAngle). (we use y-up world frame coordinate)
234 // - you can get the robot's forward direction vector from robot->forward
235 // - you can get the robot's sideways direction vector by RBGlobals::worldUp.cross(robot->forward)
236
237 // get rotation matrix around y axis with angle 'headingAngle'
238 Matrix rot = Eigen::AngleAxisd(headingAngle, Eigen::Vector3d::UnitY()).toRotationMatrix();
239
240 //calculate new position and angle
241 pos.y = targetbFrameHeight + ground.get_height(pos.x, pos.z);
242 pos = pos + dt * (rot * V3D(0, 0, 1) * vForward + rot * RBGlobals::worldUp.cross(V3D(0, 0, 1)) * vSideways);
243 headingAngle = headingAngle + dt * turningSpeed;
244
245 t += dt;
246 }
247 }

315 void populateFootstepPlan(FootstepPlan& fsp, const LimbMotionProperties& lmProps, ContactPlanManager* cpm, double groundHeight = 0) {
316 fsp.cpm = cpm;
317 double tTiny = 0.0001;
318 for (uint i = 0; i < robot->getLimbCount(); i++) {
319 const auto& limb = robot->getLimb(i);
320 fsp.footSteps[limb].clear();
321 ContactPhaseInfo cpi = cpm->getCPInformationFor(limb, tStart);
322 double t = tStart;
323 if (cpi.isStance()) {
324 P3D pos = limb->getEEWorldPos();
325 pos.y = groundHeight + limb->ee->radius * lmProps.contactSafetyFactor; // account for the size of the ee
326 fsp.footSteps[limb].push_back(PlannedLimbContact(tStart, tStart + cpi.getTimeLeft(), pos, true));
327 t = tStart + cpi.getTimeLeft() + tTiny;
328 }
329
330 while (t < tEnd) {
331 ContactPhaseInfo cpiSwing = cpm->getCPInformationFor(limb, t);
332 //when we get here, at time t, the foot is in swing, so find the start of the next contact phase
333 if (cpiSwing.isSwing() == false) {
334 Logger::consolePrint(
335 "ERROR, ERROR, at this point the limb should be in "
336 "swing, but isn't");
337 assert(cpiSwing.isStance());
338 }
339
340 t += cpiSwing.getTimeLeft();
341 ContactPhaseInfo cpiStance = cpm->getCPInformationFor(limb, t + tTiny);
342 if (cpiStance.isStance() == false) {
343 Logger::consolePrint(
344 "ERROR, ERROR, at this point the limb should be in "
345 "stance, but isn't");
346 assert(cpiStance.isSwing());
347 }
348
349 double tStart = t;
350 double tEnd = tStart + cpiStance.getDuration();
351 t = tEnd + tTiny;
352
353 //this is the moment in time where we'd like the limb to be right under its hip
354 double tMidStance = tStart + cpiStance.getDuration() * lmProps.ffStancePhaseForDefaultStepLength;
355 //so, compute the location of the body frame at that particular moment in time...
356 double bFrameHeadingAngle = bFrameHeadingTrajectory.evaluate_linear(tMidStance);
357 P3D bFramePos = P3D() + bFramePosTrajectory.evaluate_linear(tMidStance);
358
359 V3D defaultEEOffset = limb->defaultEEOffset;
360 defaultEEOffset[0] *= lmProps.stepWidthOffsetX;
361 defaultEEOffset[2] *= lmProps.stepWidthOffsetZ;
362
363 P3D pos = bFramePos + getRotationQuaternion(bFrameHeadingAngle, V3D(0, 1, 0)) * defaultEEOffset;
364
365 pos.y = limb->ee->radius * lmProps.contactSafetyFactor; // account for the size of the ee
366
367 fsp.footSteps[limb].push_back(PlannedLimbContact(tStart, tEnd, pos, false));
368 }
369 }
