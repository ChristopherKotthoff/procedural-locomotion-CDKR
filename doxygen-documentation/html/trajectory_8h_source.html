185 T evaluate_catmull_rom(double t, bool equalEndpointSlopes = false) const {
186 int size = (int)tValues.size();
187 if (tValues.size() == 0)
188 return T(0);
189 if (t <= tValues[0])
190 return values[0];
191 if (t >= tValues[size - 1])
192 return values[size - 1];
193 int index = getFirstLargerIndex(t);
194
195 // now that we found the interval, get a value that indicates how far we
196 // are along it
197 t = (t - tValues[index - 1]) / (tValues[index] - tValues[index - 1]);
198
199 // approximate the derivatives at the two ends
200
201 T p1 = values[index - 1];
202 T p2 = values[index];
203
204 T m1 = getSlopeEstimateAtKnot(index - 1, equalEndpointSlopes) * (tValues[index] - tValues[index - 1]);
205 T m2 = getSlopeEstimateAtKnot(index, equalEndpointSlopes) * (tValues[index] - tValues[index - 1]);
206
207 double t2, t3;
208 t2 = t * t;
209 t3 = t2 * t;
210
211 // and now perform the interpolation using the four hermite basis
212 // functions from wikipedia
213 return p1 * (2 * t3 - 3 * t2 + 1) + m1 * (t3 - 2 * t2 + t) + p2 * (-2 * t3 + 3 * t2) + m2 * (t3 - t2);
214 }
