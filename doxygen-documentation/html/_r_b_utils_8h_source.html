81inline void decomposeRotation(const Quaternion &PqC, Quaternion &Pq, Quaternion &qC, const V3D &v) {
82 // we need to compute v in P's coordinates
83 V3D vP = PqC * v;
84 // compute the rotation that alligns the vector v in the two coordinate
85 // frames (P and T - remember that v has the same coordinates in C and in T)
86 V3D rotAxis = vP.cross(v);
87 rotAxis.normalize();
88 double rotAngle = -safeACOS(vP.dot(v));
89
90 Pq = getRotationQuaternion(rotAngle, rotAxis);
91 // now qB = qAinv * PqC
92 qC = Pq.inverse() * PqC;
93}

98inline Quaternion decomposeRotation(const Quaternion &PqC, const V3D &v) {
99 Quaternion Pq = Quaternion::Identity(), qC = Quaternion::Identity();
100 decomposeRotation(PqC, Pq, qC, v);
101 return qC;
102}

173inline Quaternion updateRotationGivenAngularVelocity(const Quaternion &q, const V3D &angularVelocity, double dt) {
174 // q_p = rot(w, dt) * q
175 double angularVelocityMagnitude = angularVelocity.norm();
176 // avoid divide by close to zero...
177 if (angularVelocityMagnitude > 1e-10) {
178 // compute the rotation that would have happened during dt, and add it
179 // (pre) to the current orientation
180 double netRotationAngle = angularVelocityMagnitude * dt;
181 V3D rotationAxis = angularVelocity.normalized();
182 return getRotationQuaternion(netRotationAngle, rotationAxis) * q;
183 }
184 return q;
185}

191inline V3D estimateAngularVelocity(const Quaternion &qStart, const Quaternion &qEnd, double dt) {
192 // qEnd = rot(w_p, dt) * qStart
193 Quaternion qRot = qEnd * qStart.inverse();
194
195 V3D rotAxis = Vector3d(qRot.vec());
196 if (rotAxis.norm() < 1e-10)
197 return V3D(0, 0, 0);
198 rotAxis.normalize();
199 double rotAngle = getRotationAngle(qRot, rotAxis);
200
201 // this rotation angle is the result of applying the angular velocity for
202 // some time dt...
203 return rotAxis * rotAngle / dt;
204}
