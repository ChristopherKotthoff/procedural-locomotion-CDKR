81 // If ClosestPtOnRay is not nullptr, it will be set to the point on the ray
82 // that is closest to the segment
83 double getDistanceToSegment(const P3D &p1, const P3D &p2, P3D *closestPtOnRay) const {
84 // (c) Dan Sunday, http://geomalgorithms.com/a07-_distance.html
85 V3D u = V3D(p1, origin) + dir * 10000;
86 V3D v(p1, p2);
87 V3D w(p1, origin);
88
89 double a = u.squaredNorm();
90 double b = u.dot(v);
91 double c = v.squaredNorm();
92 double d = u.dot(w);
93 double e = v.dot(w);
94 double denom = a * c - b * b;
95
96 double sN, sD = denom;
97 double tN, tD = denom;
98
99 if (denom <= 1e-10) {
100 // lines are almost parallel
101 sN = 0;
102 sD = 1;
103 tN = e;
104 tD = c;
105 } else {
106 sN = (b * e - c * d);
107 tN = (a * e - b * d);
108 if (sN < 0) {
109 // the closest point is behind the ray's starting point
110 sN = 0;
111 tN = e;
112 tD = c;
113 }
114 }
115
116 if (tN < 0) {
117 // the closest point is behind t1
118 tN = 0;
119 if (-d < 0)
120 sN = 0;
121 else if (-d > a)
122 sN = sD;
123 else {
124 sN = -d;
125 sD = a;
126 }
127 } else if (tN > tD) {
128 // the closest point is behind t2
129 tN = tD;
130 if ((-d + b) < 0)
131 sN = 0;
132 else if ((-d + b) > a)
133 sN = sD;
134 else {
135 sN = -d + b;
136 sD = a;
137 }
138 }
139
140 double sc = (abs(sN) < 1e-10) ? 0 : sN / sD;
141 double tc = (abs(tN) < 1e-10) ? 0 : tN / tD;
142
143 if (closestPtOnRay != nullptr)
144 *closestPtOnRay = origin + dir * (sc * 10000);
145
146 return (w + u * sc - v * tc).norm();
147 }
148
149 // returns the point on the ray that is closest to c.
150 // If ClosestPtOnRay is not nullptr, it will be set to the point on the ray
151 // that is closest to c
152 double getDistanceToPoint(const P3D &p, P3D *closestPtOnRay) const {
153 double d = dir.dot(V3D(origin, p));
154 P3D res = origin;
155 if (d > 0)
156 res = getPointAt(d);
157
158 if (closestPtOnRay)
159 *closestPtOnRay = res;
160
161 return V3D(p, res).norm();
162 }

164 // returns the smallest distance between the line segment and the ray.
165 // and if ClosestPtOnRay is not NULL, it will be set to the point on the ray
166 // that is closest to c
167 double getDistanceToPlane(const Plane &plane, P3D *closestPtOnRay) const {
168 // check to see if the ray is parallel to the plane...
169 if (fabs(dir.dot(plane.n)) < 10e-10) {
170 if (closestPtOnRay)
171 *closestPtOnRay = origin;
172 return fabs(plane.getSignedDistanceToPoint(origin));
173 }
174
175 // we know that p = origin + t * direction lies on the plane, which
176 // means that dot product between plane p and p, dotted with n is 0...
177 V3D tmpV(origin, plane.p);
178 double t = tmpV.dot(plane.n) / dir.dot(plane.n);
179
180 // check the solution now...
181 assert(IS_ZERO(V3D(plane.p, origin + dir * t).dot(plane.n)));
182
183 if (t < 0) {
184 if (closestPtOnRay)
185 *closestPtOnRay = origin;
186 return fabs(plane.getSignedDistanceToPoint(origin));
187 }
188
189 if (closestPtOnRay)
190 *closestPtOnRay = origin + dir * t;
191
192 return t;
193 }
